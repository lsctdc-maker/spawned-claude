# 상세페이지 원고 자동화 — Claude Code 구현 가이드

## 목표
Spawned Claude(상주 프로세스) 방식으로
입력 → 분석 → 크롤링 → 구조 분석 → 원고 작성 → 자동 검토 루프까지
이미지 제외, 원고 단계만 완전 자동화

## 핵심 설계 원칙: Context 유지

기존 방식과 다르게, 모든 단계가 하나의 `context` 객체를 공유함.
Claude를 매번 새로 호출하되, **이전 단계 결과를 항상 포함**해서 전달.

```python
# context.py

class ProjectContext:
    def __init__(self, raw_manuscript: str):
        self.raw_manuscript = raw_manuscript     # 원본 원고
        self.product_info = {}                   # STEP 1 결과
        self.reference_pages = []                # STEP 2 결과
        self.structure_pattern = {}              # STEP 3 결과
        self.draft = ""                          # STEP 4 결과
        self.review_history = []                 # 검토 이력 (실패 이유 누적)
        self.final_manuscript = ""               # 최종 완성 원고

    def to_prompt_context(self) -> str:
        """현재까지 파악한 모든 맥락을 프롬프트용 문자열로 변환"""
        parts = [f"[원본 원고]\n{self.raw_manuscript}"]

        if self.product_info:
            parts.append(f"[상품 분석 결과]\n{self.product_info}")

        if self.structure_pattern:
            parts.append(f"[레퍼런스 구조 패턴]\n{self.structure_pattern}")

        if self.review_history:
            failures = "\n".join([
                f"- {r['round']}차 시도: {r['score']}점 / 실패 이유: {r['reason']}"
                for r in self.review_history
            ])
            parts.append(f"[이전 검토 실패 이력 — 반드시 반영할 것]\n{failures}")

        return "\n\n".join(parts)
```

## 각 단계 구현

### STEP 1 — 원고 분석

```python
# steps/analyze.py
import anthropic

def analyze_manuscript(ctx, client):
    print("▶ STEP 1: 원고 분석 중...")

    prompt = f"""
다음 상품 원고를 분석해서 JSON으로 반환해줘.

{ctx.raw_manuscript}

반환 형식:
{{
  "product_name": "상품명",
  "category": "카테고리",
  "target": "타겟 고객",
  "key_features": ["특징1", "특징2", "특징3"],
  "tone": "원고 톤 (감성적/전문적/친근한 등)",
  "keywords": ["핵심 키워드1", "키워드2", "키워드3"]
}}
"""

    response = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=1000,
        messages=[{"role": "user", "content": prompt}]
    )

    import json
    ctx.product_info = json.loads(response.content[0].text)
    print(f"   완료: {ctx.product_info['product_name']} / 타겟: {ctx.product_info['target']}")
    return ctx
```

### STEP 2 — 핀터레스트 크롤링

```python
# steps/crawl.py
from playwright.sync_api import sync_playwright

def crawl_pinterest(ctx):
    print("▶ STEP 2: 핀터레스트 크롤링 중...")

    # STEP 1 결과에서 키워드 가져옴 (맥락 연결)
    keywords = ctx.product_info.get("keywords", [])
    search_query = " ".join(keywords[:2]) + " 상세페이지"

    results = []

    with sync_playwright() as p:
        browser = p.chromium.launch(headless=True)
        page = browser.new_page()
        page.goto(f"https://www.pinterest.com/search/pins/?q={search_query}")
        page.wait_for_timeout(3000)

        # 상위 10개 핀 텍스트 수집
        pins = page.query_selector_all("[data-test-id='pin']")
        for pin in pins[:10]:
            text = pin.inner_text()
            if text.strip():
                results.append(text.strip())

        browser.close()

    ctx.reference_pages = results
    print(f"   완료: {len(results)}개 레퍼런스 수집")
    return ctx
```

### STEP 3 — 구조 분석

```python
# steps/structure.py

def analyze_structure(ctx, client):
    print("▶ STEP 3: 구조 분석 중...")

    # STEP 1 + STEP 2 결과 모두 포함 (맥락 연결)
    prompt = f"""
{ctx.to_prompt_context()}

[수집된 레퍼런스 페이지 텍스트]
{chr(10).join(ctx.reference_pages[:5])}

위 레퍼런스들의 공통 구조 패턴을 분석해서 JSON으로 반환해줘.

반환 형식:
{{
  "sections": [
    {{"order": 1, "name": "섹션명", "role": "역할", "copy_pattern": "카피 패턴"}},
    ...
  ],
  "hook_style": "후킹 방식",
  "cta_style": "CTA 방식"
}}
"""

    response = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=1500,
        messages=[{"role": "user", "content": prompt}]
    )

    import json
    ctx.structure_pattern = json.loads(response.content[0].text)
    print(f"   완료: {len(ctx.structure_pattern['sections'])}개 섹션 구조 도출")
    return ctx
```

### STEP 4 + 검토 루프 — 원고 작성 & 자동 검토

```python
# steps/write.py

def write_and_review(ctx, client, max_attempts=5):
    print("▶ STEP 4: 원고 작성 + 자동 검토 루프 시작")

    for attempt in range(1, max_attempts + 1):
        print(f"\n   [{attempt}차 시도]")

        # 원고 작성
        ctx.draft = _write_draft(ctx, client)

        # 검토 점수
        score, reason, suggestions = _review_draft(ctx, client)
        print(f"   점수: {score}/10 — {reason}")

        if score >= 8:
            print(f"   ✅ {score}점 통과! 원고 확정")
            ctx.final_manuscript = ctx.draft
            return ctx

        # 실패 이력 누적 (다음 시도에 반영됨)
        ctx.review_history.append({
            "round": attempt,
            "score": score,
            "reason": reason,
            "suggestions": suggestions,
            "draft": ctx.draft
        })
        print(f"   ❌ 8점 미달 — 재작성 (이유: {reason})")

    print("   ⚠️ 최대 시도 횟수 도달 — 마지막 버전으로 저장")
    ctx.final_manuscript = ctx.draft
    return ctx


def _write_draft(ctx, client):
    """원고 작성 — 전체 맥락 포함"""
    prompt = f"""
{ctx.to_prompt_context()}

위 맥락을 바탕으로 상세페이지 원고를 작성해줘.

구조는 반드시 아래 패턴을 따를 것:
{ctx.structure_pattern}

요구사항:
- 타겟: {ctx.product_info.get('target')}
- 톤: {ctx.product_info.get('tone')}
- 각 섹션별 헤드카피 + 서브카피 + 본문 포함
- 이전 검토에서 지적된 문제점 반드시 수정할 것

원고만 출력할 것 (설명 없이)
"""

    response = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=3000,
        messages=[{"role": "user", "content": prompt}]
    )
    return response.content[0].text


def _review_draft(ctx, client):
    """원고 검토 — 점수 + 이유 + 개선 제안 반환"""
    prompt = f"""
{ctx.to_prompt_context()}

[검토할 원고]
{ctx.draft}

이 원고를 아래 기준으로 10점 만점 채점해줘.

채점 기준:
- 후킹력 (첫 문장에서 멈추게 하는가): 3점
- 구조 완성도 (섹션 흐름이 자연스러운가): 3점
- 타겟 적합성 (타겟 고객에게 와닿는가): 2점
- CTA 명확성 (구매 유도가 명확한가): 2점

반드시 JSON으로만 반환:
{{
  "score": 점수,
  "reason": "낮은 이유 한 줄 요약",
  "suggestions": ["개선점1", "개선점2"]
}}
"""

    response = client.messages.create(
        model="claude-opus-4-5",
        max_tokens=500,
        messages=[{"role": "user", "content": prompt}]
    )

    import json
    result = json.loads(response.content[0].text)
    return result["score"], result["reason"], result["suggestions"]
```

### 메인 파이프라인

```python
# pipeline.py
import anthropic
from context import ProjectContext
from steps.analyze import analyze_manuscript
from steps.crawl import crawl_pinterest
from steps.structure import analyze_structure
from steps.write import write_and_review

def run(manuscript_text: str):
    client = anthropic.Anthropic()

    # 맥락 객체 생성 — 여기가 Spawned 방식의 핵심
    ctx = ProjectContext(manuscript_text)

    ctx = analyze_manuscript(ctx, client)   # STEP 1
    ctx = crawl_pinterest(ctx)              # STEP 2
    ctx = analyze_structure(ctx, client)    # STEP 3
    ctx = write_and_review(ctx, client)     # STEP 4 + 검토 루프

    return ctx.final_manuscript
```

```python
# main.py
from pipeline import run

manuscript = """
여기에 원고 붙여넣기
상품명, 특징, 타겟 등 원본 내용
"""

result = run(manuscript)

print("\n" + "="*50)
print("최종 완성 원고")
print("="*50)
print(result)

# 파일로도 저장
with open("output_manuscript.txt", "w", encoding="utf-8") as f:
    f.write(result)
```

## 기존 방식과 차이점 요약

| | 기존 단발 API | 이 구현 (Spawned) |
|--|--|--|
| 맥락 전달 | 각 단계마다 수동으로 프롬프트 재구성 | `ctx.to_prompt_context()`로 자동 포함 |
| 실패 학습 | 재시도해도 왜 실패했는지 모름 | `review_history`에 누적, 다음 시도에 반영 |
| 타겟 연결 | STEP 1 타겟을 STEP 4에서 다시 입력 | ctx 객체로 자동 흐름 |
| 검토 루프 | 수동으로 점수 보고 다시 실행 | 8점 이상 될 때까지 자동 반복 |

## Claude Code에서 실행하는 방법

```bash
# Claude Code 터미널에서
claude

# 프롬프트 입력:
"detail-page-writer 폴더 만들고 위 구조대로 파일 생성해줘.
main.py에 아래 원고 넣고 전체 파이프라인 실행해봐.
[원고 내용 붙여넣기]"
```

Claude Code가 파일 생성 → 설치 → 실행 → 결과 확인을 알아서 처리함.
